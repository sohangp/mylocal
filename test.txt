kubectl apply -f k8s

---

apiVersion: v1
kind: Namespace
metadata:
  name: redis-namespace

---
apiVersion: v1
kind: ConfigMap
metadata:
  name: redis-config
  namespace: redis-namespace
data:
  REDIS_HOST: redis-service.redis-namespace.svc.cluster.local
  REDIS_PORT: "6379"

---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: my-go-microservice
  namespace: redis-namespace
spec:
  replicas: 1
  selector:
    matchLabels:
      app: my-go-microservice
  template:
    metadata:
      labels:
        app: my-go-microservice
    spec:
      containers:
      - name: my-go-microservice
        image: my-go-microservice:latest
        env:
        - name: REDIS_HOST
          valueFrom:
            configMapKeyRef:
              name: redis-config
              key: REDIS_HOST
        - name: REDIS_PORT
          valueFrom:
            configMapKeyRef:
              name: redis-config
              key: REDIS_PORT
        ports:
        - containerPort: 8080

---
apiVersion: v1
kind: Service
metadata:
  name: redis-service
  namespace: redis-namespace
spec:
  type: ClusterIP
  ports:
  - name: redis
    port: 6379
    targetPort: 6379
  selector:
    app: redis
